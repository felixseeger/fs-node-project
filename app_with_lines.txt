 509 │   const handleImagePaste = useCallback(async (event: ClipboardEvent) => {
 510 │     if (!event.clipboardData || !rfInstance) return;

 514 │       x: window.innerWidth / 2,
 515 │       y: window.innerHeight / 2
 516 │     });

 518 │     const files = Array.from(event.clipboardData.files);
 519 │     
 520 │     if (files.length > 0) {
 521 │       // Handle file paste (e.g. copying from Finder/Explorer or screenshot)
 522 │       // Mac screenshots might put both PNG and TIFF in the clipboard, we want to deduplicate 
 523 │       // if they seem to be the same image (same size & timestamp) or just take the first one if it's a generic screenshot.
 524 │       // But if user copies multiple distinct files, we want to paste all of them.
 525 │       const processedFiles = new Set<string>();
 526 │ 
 527 │       for (const file of files) {
 528 │         const hasImageExtension = /\.(jpeg|jpg|png|gif|webp|svg|tiff|bmp)$/i.test(file.name);
 529 │         const isExplicitlyNotImage = file.type !== '' && !file.type.startsWith('image/');
 530 │         
 531 │         if (!isExplicitlyNotImage || hasImageExtension) {
 532 │           // simple deduplication based on size and generic "image" name
 533 │           const fileKey = `${file.name}-${file.size}`;
 534 │           if (processedFiles.has(fileKey)) continue;
 535 │           processedFiles.add(fileKey);
 536 │ 
 537 │           const node = await createMediaNodeFromFile(file, { x: 0, y: 0 });
 538 │           if (node) newNodes.push(node);
 539 │         }
 540 │       }
